Массивы в JavaScript - это именованный набор не строго типизированный переменных, расположенных в памяти непосредственно друг за другом, доступ к которым осуществляется по индексу.

{% highlight JavaScript %} var d = ["Hello", 34, true]; {% endhighlight %}

Массив - обьект, у которого имеется свойство length, это свойство хранит в себе длину массива.

Это свойство удаляет и/или добавляет елементы массива в зависимости от присвоения неотрицательного целого числа.

{% highlight JavaScript %} var array = ['h', 'e', 'l', 'l', 'o']; array.length = 7; for (var i = 0; i < array.length; i++) { document.write(array[i] + "
"); } document.write("
"); {% endhighlight %}

Двумерная матрица

Массивы бывают многомерными, для этого в каждый элемент массива следует расположить еще один массив. Таким образом возможно реализовать n-мерный массив.

{% highlight JavaScript %} var table = new Array(10); for (var i = 0; i < table.length; i++) { // В таблице 10 строк table[i] = new Array(10); // В каждой строке 10 столбцов } // Инициализация массива. for (var row = 0; row < table.length; row++) { for (var col = 0; col < table[row].length; col++) { if ((row + col) % 2 == 1) table[row][col] = "1"; else table[row][col] = "0"; //document.write(table[row][col]); //вывод заполненой ячейки сразу } // document.write("
");//разрыв строки } //table[8][1] = ""; for (var row = 0; row < table.length; row++) { for (var col = 0; col < table[row].length; col++) { document.write(table[row][col]); } document.write("
"); } var prod = table[8][1]; document.write("
" + prod); {% endhighlight %}

Добавление в массив

concat() - создает и возвращает новый массив, содержащий элементы исходного массива, для которого был вызван метод concat(), и значения всех аргументов, переданных методу concat().

{% highlight JavaScript %} var a = [1, 2, 3]; var input = prompt("Введите новый элемент массива:", ""); while (input != null) { a = a.concat(input); input = prompt("Введите новый элемент массива:", ""); } document.write(a.join("
")); {% endhighlight %}

Метод slice()

slice() - Возвращает фрагмент или подмассив указанного массива.

{% highlight JavaScript %} var a = [1, 2, 3, 4, 5, 6, 7]; // slice(начальный индекс (включительно), конечный индекс (не включительно)); var star = 2; var end = 6; var temp = a.slice(star, end); document.write("

" + temp); // slice(начальный индекс); - возвращает подмассив от указанного индекса до конца массива. temp = a.slice(3); // 4, 5, 6, 7 document.write("

" + temp); // Если аргумент отрицательный - он задает позицию относительно конца массива. temp = a.slice(1, -2); // 2, 3, 4, 5, 6 document.write("

" + temp); {% endhighlight %}

Метод splice()

Метод splice() - универсальный метод для вставки и удаления элементов из массива, возвращает массив из удаленных элементов.

{% highlight JavaScript %} var a = [1, 2, 3, 4, 5, 6, 7]; document.write("

Исходный массив: " + a); // УДАЛЕНИЕ. /* splice (начальный индекс, сколько элементов удалить); Если второй аргумент опущен, удаляются элементы массива до конца массива */ var d = a.splice(1, 3); // Будут удалены: 2, 3, 4 document.write("

Удалены: " + d); document.write("

Остались: " + a); // ВСТАВКА. /*(начальный индекс = 1, сколько элементов удалить = 0, значения для вставки с начальниндекса = 2, 3, 4); */ a.splice(1, 0, 2, 3, 4); document.write("

После вставки: " + a); {% endhighlight %}

Метод push()

push() - добавление одного или нескольких элементов в конец и массива возвращает его новую длину. (возвращаемое значение - количество элементов в новом массиве)

{% highlight JavaScript %} var stack = ["January", "February", "March"]; var lenght = stack.push("April"); document.write("

После добавления: " + stack.join(" ")); document.write("

Размер массива: " + lenght); {% endhighlight %}

Метод pop()

pop() - выполняет удаление последнего элемента массива, уменьшает длину массива. (возвращаемое значение - удаленный элемент массива)

{% highlight JavaScript %} var stack = ["January", "February", "March", "April", "May", "June"]; document.write("

Исходный: " + stack.join(" ")); document.write("

Размер массива: " + stack.length); document.write("

"); var lenght = stack.pop(); document.write("

После pop: " + stack.join(" ")); document.write("

Удаленный элемент: " + lenght); document.write("

Размер массива после удаления: " + stack.length); {% endhighlight %}

Метод shift()

shift() - удаляет и возвращает первый элемент массива смещая все последующие элементы на одну позицию вниз, чтобы занять место, освободившееся в начале массива.

{% highlight JavaScript %} var arr = [1, 2, 3]; document.write("

Исходный массив: " + arr.join(" ")); document.write("

"); var value = arr.shift(); document.write("

Массив после сдвига: " + arr.join(" ")); document.write("

Выдвинутое значение: " + value); {% endhighlight %}

Метод unshift()

unshift() - добавляет элемент в начало массива. Смещает существующие элементы в сторону больших индексов для освобождения места, добавляет элемент или элементы в начало массива и возвращает новую длину массива.

{% highlight JavaScript %} var arr = [1, 2, 3]; document.write("

Исходный массив: " + arr.join(" ")); document.write("

"); arr.unshift(0); document.write("

Массив после сдвига: " + arr.join(" ")); document.write("

"); arr.unshift(-3, -2, -1); document.write("

Массив после сдвига: " + arr.join(" ")); {% endhighlight %}
